// ─── AI Generator ─────────────────────────────────────────────────────────────
async function generateArticle(topic, cat, length) {
  const catName = CATS.find(c => c.id === cat)?.label || cat;
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: `Kamu adalah penulis untuk platform edukasi Indonesia bernama CCA (Cerdas, Cerah, Asik). Gaya penulisanmu: langsung, padat, tidak menggurui, seperti teman cerdas yang menjelaskan sesuatu yang menarik. Gunakan "lo/kamu" bukan "Anda". Hindari jargon yang tidak perlu. Setiap paragraf harus ada nilainya.

Tulis artikel dalam format JSON dengan struktur ini (HANYA JSON, tidak ada teks lain):
{
  "title": "judul menarik dan spesifik",
  "sub": "subjudul 1 kalimat yang memperjelas dan menarik",
  "tag": "LABEL KATEGORI (huruf kapital, 2-3 kata)",
  "readTime": "X menit",
  "content": [
    {"type": "lead", "text": "paragraf pembuka yang langsung menarik"},
    {"type": "heading", "text": "subjudul"},
    {"type": "body", "text": "paragraf isi"},
    {"type": "callout", "text": "kutipan atau fakta menarik yang stand-alone"},
    {"type": "note", "text": "fakta tambahan yang memperkaya"},
    {"type": "closing", "text": "penutup yang meninggalkan sesuatu untuk dipikirkan"}
  ]
}`,
      messages: [{ role: "user", content: `Tulis artikel ${length === "pendek" ? "pendek (3-5 menit baca, 4-5 blok konten)" : "panjang (7-10 menit baca, 7-9 blok konten)"} tentang: "${topic}" untuk kategori ${catName}.` }],
    }),
  });
  const data = await res.json();
  const text = data.content?.[0]?.text || "{}";
  const clean = text.replace(/```json|```/g, "").trim();
  return JSON.parse(clean);
}

// ─── Components ───────────────────────────────────────────────────────────────

function ReadingBar({ pct, color }) {
  return <div style={{ position: "fixed", top: 0, left: 0, height: 3, width: `${pct}%`, background: color, zIndex: 300, transition: "width .1s" }} />;
}

function ArticleContent({ blocks, color }) {
  return (
    <div>
      {blocks.map((b, i) => {
        if (b.type === "lead") return (
          <p key={i} style={{ fontSize: 19, lineHeight: 1.82, color: "#222", borderLeft: `3px solid ${color}`, paddingLeft: 18, marginBottom: 28, fontStyle: "normal" }}>{b.text}</p>
        );
        if (b.type === "heading") return (
          <h2 key={i} style={{ fontSize: 22, fontWeight: 700, color: T.ink, marginTop: 40, marginBottom: 12, lineHeight: 1.28 }}>{b.text}</h2>
        );
        if (b.type === "body") return (
          <p key={i} style={{ fontSize: 16.5, lineHeight: 1.9, color: "#2E2E26", marginBottom: 20 }}>{b.text}</p>
        );
        if (b.type === "callout") return (
          <div key={i} style={{ margin: "32px 0", padding: "20px 24px", borderLeft: `4px solid ${color}`, background: color + "10" }}>
            <p style={{ fontSize: 18, fontStyle: "italic", color: T.ink, lineHeight: 1.65, margin: 0 }}>{b.text}</p>
          </div>
        );
        if (b.type === "note") return (
          <div key={i} style={{ margin: "24px 0", padding: "14px 18px", background: T.faint, borderRadius: 0, borderTop: `1px solid ${T.border}` }}>
            <div style={{ fontSize: 10, fontFamily: "monospace", letterSpacing: 2, color: T.muted, marginBottom: 6 }}>TAHUKAH LO?</div>
            <p style={{ fontSize: 14.5, lineHeight: 1.75, color: "#555", margin: 0 }}>{b.text}</p>
          </div>
        );
        if (b.type === "closing") return (
          <p key={i} style={{ fontSize: 16.5, lineHeight: 1.9, color: "#2E2E26", marginBottom: 20, paddingTop: 24, borderTop: `1px solid ${T.border}` }}>{b.text}</p>
        );
        return null;
      })}
    </div>
  );
}

function ArticleReader({ article, onClose }) {
  const ref = useRef(null);
  const [prog, setProg] = useState(0);
  const col = catColor(article.cat);
  const series = article.series ? SERIES.find(s => s.id === article.series) : null;

  useEffect(() => {
    window.scrollTo(0, 0);
    const el = ref.current;
    if (!el) return;
    const fn = () => setProg(Math.min((el.scrollTop / (el.scrollHeight - el.clientHeight)) * 100, 100));
    el.addEventListener("scroll", fn);
    return () => el.removeEventListener("scroll", fn);
  }, []);

  return (
    <div style={{ position: "fixed", inset: 0, zIndex: 200, background: T.white, display: "flex", flexDirection: "column" }}>
      <ReadingBar pct={prog} color={col} />

      {/* Nav */}
      <div style={{ padding: "14px 24px", borderBottom: `1px solid ${T.border}`, display: "flex", justifyContent: "space-between", alignItems: "center", background: T.white, position: "sticky", top: 0, zIndex: 10 }}>
        <button onClick={onClose} style={{ background: "none", border: "none", cursor: "pointer", fontSize: 13, color: T.muted, fontFamily: "inherit", display: "flex", alignItems: "center", gap: 6 }}>

function AIGeneratorPanel({ onAdd }) {
  const [open, setOpen] = useState(false);
  const [topic, setTopic] = useState("");
  const [cat, setCat] = useState("logika");
  const [length, setLength] = useState("pendek");
  const [loading, setLoading] = useState(false);
  const [dots, setDots] = useState(0);
  const [err, setErr] = useState("");

  useEffect(() => {
    if (loading) {
      const t = setInterval(() => setDots(d => (d + 1) % 4), 350);
      return () => clearInterval(t);
    }
  }, [loading]);

  async function generate() {
    if (!topic.trim()) return;
    setLoading(true); setErr("");
    try {
      const data = await generateArticle(topic, cat, length);
      const newArt = {
        id: Date.now(), cat, series: null, featured: false, length,
        title: data.title, sub: data.sub, tag: data.tag,
        readTime: data.readTime, content: data.content,
      };
      onAdd(newArt);
      setTopic(""); setOpen(false);
    } catch (e) {
      setErr("Gagal generate artikel. Coba lagi.");
    }
    setLoading(false);
  }

  return (
    <div style={{ marginBottom: 32 }}>
      {!open ? (
        <button onClick={() => setOpen(true)} style={{
          background: "none", border: `1px dashed ${T.border}`,
          color: T.muted, fontFamily: "monospace", fontSize: 10, letterSpacing: 2,
          padding: "10px 20px", cursor: "pointer", transition: "all .2s", width: "100%",
        }}
          onMouseEnter={e => { e.currentTarget.style.borderColor = T.ink; e.currentTarget.style.color = T.ink; }}
          onMouseLeave={e => { e.currentTarget.style.borderColor = T.border; e.currentTarget.style.color = T.muted; }}
        >
          + GENERATE ARTIKEL BARU DENGAN AI
        </button>
      ) : (
        <div style={{ border: `1px solid ${T.border}`, padding: 24, background: T.white }}>
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", marginBottom: 18 }}>
            <span style={{ fontFamily: "monospace", fontSize: 10, letterSpacing: 2, color: T.muted }}>GENERATE ARTIKEL</span>
            <button onClick={() => setOpen(false)} style={{ background: "none", border: "none", cursor: "pointer", color: T.muted, fontSize: 16 }}>✕</button>
          </div>
          <input value={topic} onChange={e => setTopic(e.target.value)}
            onKeyDown={e => e.key === "Enter" && generate()}
            placeholder="Topik artikel (mis: 'kenapa kita suka konspirasi')"
            style={{ width: "100%", background: T.faint, border: "none", outline: "none", fontFamily: "Georgia,serif", fontSize: 15, color: T.ink, padding: "12px 14px", marginBottom: 12 }}
          />
          <div style={{ display: "flex", gap: 8, marginBottom: 16, flexWrap: "wrap" }}>
            {CATS.slice(1).map(c => (
              <button key={c.id} onClick={() => setCat(c.id)} style={{
                background: cat === c.id ? c.color : "none",
                border: `1px solid ${cat === c.id ? c.color : T.border}`,
                color: cat === c.id ? "#fff" : T.muted,
                fontFamily: "monospace", fontSize: 9, letterSpacing: 1, padding: "5px 12px", cursor: "pointer", transition: "all .15s",
              }}>{c.label}</button>
            ))}
          </div>
          <div style={{ display: "flex", gap: 8, marginBottom: 18 }}>
            {["pendek", "panjang"].map(l => (
              <button key={l} onClick={() => setLength(l)} style={{
                background: length === l ? T.ink : "none",
                border: `1px solid ${length === l ? T.ink : T.border}`,
                color: length === l ? "#fff" : T.muted,
                fontFamily: "monospace", fontSize: 9, letterSpacing: 1, padding: "5px 14px", cursor: "pointer",
              }}>{l}</button>
            ))}
          </div>
          {err && <p style={{ color: "#C44", fontFamily: "monospace", fontSize: 11, marginBottom: 12 }}>{err}</p>}
          <button onClick={generate} disabled={loading || !topic.trim()} style={{
            background: loading ? T.faint : T.ink, border: "none", color: loading ? T.muted : "#F5F5EF",
            fontFamily: "monospace", fontSize: 10, letterSpacing: 2, padding: "11px 24px",
            cursor: loading ? "wait" : "pointer", transition: "all .2s", width: "100%",
          }}>
            {loading ? `SEDANG MENULIS${".".repeat(dots)}` : "GENERATE →"}
          </button>
        </div>
      )}
    </div>
  );
}
